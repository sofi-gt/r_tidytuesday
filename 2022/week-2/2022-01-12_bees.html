Create plot for the United States

us_colonies <- colony %>% 
  filter(state == "United States") %>% 
  group_by(year) %>% 
  summarise(
    colonies_lost = mean(colony_lost, na.rm = TRUE),
    colonies_added = mean(colony_added, na.rm = TRUE)
  ) %>% 
  pivot_longer(cols = c(
    colonies_lost, 
    colonies_added))
